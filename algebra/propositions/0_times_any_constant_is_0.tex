\begin{proposition}{0 times any Constant is 0}{0_times_any_constant_is_0}
\[
c 0 = 0
\]
\end{proposition}
\begin{proof}
\begin{gather*}
    c0 = c \left( 0 +  0 \right) =  c0 +  c0\\
    \Updownarrow \\
    c0 -  c0 =  c0 +  c0 -  c0\\
    \Updownarrow \\
    0 =  c0\\
    \Updownarrow \\
    c0 = 0
\end{gather*}
\end{proof}
