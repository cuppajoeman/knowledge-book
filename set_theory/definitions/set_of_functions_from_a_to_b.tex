\begin{definition}{Set of Functions from A to B}{set_of_functions_from_a_to_b}
    Let \( A, B \) be sets, then we define the set of all functions with domain
    \( A \) and codomain \( B \) as
    \[
    f\substack{B\\\uparrow\\A} \stackrel{\mathtt{D}}{=} \left\{ f \mid f : A \to
    B\right\} 
    \]
\end{definition}
