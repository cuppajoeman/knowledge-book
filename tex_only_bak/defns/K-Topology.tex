\begin{defn*}{K-Topology}
    Define $ \mathcal{ K }  = \left\{ \frac{1}{n}: n \in  \mathbb{N} ^{ + }  \right\}  $, and $ \mathcal{ B } =  \left\{ \left( a, b \right): a < b \right\} \cup \left\{ \left( a, b \right) \setminus \mathcal{ K } : a < b \right\}$ then the topology generated by $ \mathcal{ B }  $ is called the K-Topology
\end{defn*}
