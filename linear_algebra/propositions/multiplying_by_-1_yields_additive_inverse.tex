\begin{proposition}{Multiplying by -1 Yields Additive
Inverse}{multiplying_by_-1_yields_additive_inverse}
For any \( v \in  V \) 
\[
\left( - 1 \right) v =  - v
\]
\end{proposition}
\begin{proof}
    \[
    v +  \left( - 1 \right) v =  1 v +  \left( - 1 \right) v =  \left( 1 +  - 1
    \right) v =  0v = 0
    \]
\end{proof}
